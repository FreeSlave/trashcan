26 /**
27  * Flags to rule the trashing behavior.
28  *
29  * $(BLUE Valid only for freedesktop environments).
30  *
31  * See_Also: $(LINK2 https://specifications.freedesktop.org/trash-spec/trashspec-1.0.html, Trash specification).
32  */
33 enum TrashOptions : int
34 {
35     /**
36      * No options. Just move file to user home trash directory
37      * not paying attention to partition where file resides.
38      */
39     none = 0,
40     /**
41      * If file that needs to be deleted resides on non-home partition
42      * and top trash directory ($topdir/.Trash/$uid) failed some check,
43      * fallback to user top trash directory ($topdir/.Trash-$uid).
44      *
45      * Makes sense only in conjunction with $(D useTopDirs).
46      */
47     fallbackToUserDir = 1,
48     /**
49      * If file that needs to be deleted resides on non-home partition
50      * and checks for top trash directories failed,
51      * fallback to home trash directory.
52      *
53      * Makes sense only in conjunction with $(D useTopDirs).
54      */
55     fallbackToHomeDir = 2,
56 
57     /**
58      * Whether to use top trash directories at all.
59      *
60      * If no $(D fallbackToUserDir) nor $(D fallbackToHomeDir) flags are set,
61      * and file that needs to be deleted resides on non-home partition,
62      * and top trash directory ($topdir/.Trash/$uid) failed some check,
63      * exception will be thrown. This can be used to report errors to administrator or user.
64      */
65     useTopDirs = 4,
66 
67     /**
68      * Whether to check presence of 'sticky bit' on $topdir/.Trash directory.
69      *
70      * Makes sense only in conjunction with $(D useTopDirs).
71      */
72     checkStickyBit = 8,
73 
74     /**
75      * All flags set.
76      */
77     all = (TrashOptions.fallbackToUserDir | TrashOptions.fallbackToHomeDir | TrashOptions.checkStickyBit | TrashOptions.useTopDirs)
78 }

189 /**
190  * Move file or directory to trash can.
191  * Params:
192  *  path = Path of item to remove. Must be absolute.
193  *  options = Control behavior of trashing on freedesktop environments.
194  * Throws:
195  *  Exception when given path is not absolute or does not exist or some error occured during operation.
196  */
197 @trusted void moveToTrash(string path, TrashOptions options = TrashOptions.all)
198 {
199     if (!path.isAbsolute) {
200         throw new Exception("Path must be absolute");
201     }
202     if (!path.exists) {
203         throw new Exception("Path does not exist");
204     }
205 
206     version(Windows) {
207         import core.sys.windows.shellapi;
208         import core.sys.windows.winbase;
209         import core.stdc.wchar_;
210         import std.windows.syserror;
211         import std.utf;
212 
213         SHFILEOPSTRUCTW fileOp;
214         fileOp.wFunc = FO_DELETE;
215         fileOp.fFlags = FOF_SILENT | FOF_NOCONFIRMATION | FOF_NOERRORUI | FOF_NOCONFIRMMKDIR | FOF_ALLOWUNDO;
216         auto wFileName = (path ~ "\0\0").toUTF16();
217         fileOp.pFrom = wFileName.ptr;
218         int r = SHFileOperation(&fileOp);
219         if (r != 0) {
220             import std.format;
221             throw new Exception(format("SHFileOperation failed with error code %d", r));
222         }
223     } else version(OSX) {
224         void* handle = dlopen("CoreServices.framework/Versions/A/CoreServices", RTLD_NOW | RTLD_LOCAL);
225         if (handle !is null) {
226             scope(exit) dlclose(handle);
227 
228             auto ptrFSPathMakeRefWithOptions = cast(typeof(&_dummy_FSPathMakeRefWithOptions))dlsym(handle, "FSPathMakeRefWithOptions");
229             if (ptrFSPathMakeRefWithOptions is null) {
230                 throw new Exception(fromStringz(dlerror()).idup);
231             }
232 
233             auto ptrFSMoveObjectToTrashSync = cast(typeof(&_dummy_FSMoveObjectToTrashSync))dlsym(handle, "FSMoveObjectToTrashSync");
234             if (ptrFSMoveObjectToTrashSync is null) {
235                 throw new Exception(fromStringz(dlerror()).idup);
236             }
237 
238             FSRef source;
239             enforce(ptrFSPathMakeRefWithOptions(toStringz(path), 1, &source, null) == 0, "Could not make FSRef from path");
240             FSRef target;
241             enforce(ptrFSMoveObjectToTrashSync(&source, &target, 0) == 0, "Could not move path to trash");
242         } else {
243             throw new Exception(fromStringz(dlerror()).idup);
244         }
245     } else {
246         static if (isFreedesktop) {
247             string dataPath = xdgDataHome(null, true);
248             if (!dataPath.length) {
249                 throw new Exception("Could not access data folder");
250             }
251             dataPath = dataPath.absolutePath;
252 
253             string trashBasePath;
254             bool usingTopdir = false;
255             string fileTopDir;
256 
257             if ((options & TrashOptions.useTopDirs) != 0) {
258                 string dataTopDir = topDir(dataPath);
259                 fileTopDir = topDir(path);
260 
261                 enforce(fileTopDir.length, "Could not get topdir of file being trashed");
262                 enforce(dataTopDir.length, "Could not get topdir of home data directory");
263 
264                 if (dataTopDir != fileTopDir) {
265                     try {
266                         string diskTrash = checkDiskTrash(fileTopDir, (options & TrashOptions.checkStickyBit) != 0);
267                         trashBasePath = ensureUserTrashSubdir(diskTrash);
268                         usingTopdir = true;
269                     } catch(Exception e) {
270                         try {
271                             if ((options & TrashOptions.fallbackToUserDir) != 0) {
272                                 trashBasePath = ensureUserTrashDir(fileTopDir);
273                                 usingTopdir = true;
274                             } else {
275                                 throw e;
276                             }
277                         } catch(Exception e) {
278                             if (!(options & TrashOptions.fallbackToHomeDir)) {
279                                 throw e;
280                             }
281                         }
282                     }
283                 }
284             }
285 
286             if (trashBasePath is null) {
287                 trashBasePath = ensureDirExists(buildPath(dataPath, "Trash"));
288             }
289             enforce(trashBasePath.length, "Could not access base trash folder");
290 
291             string trashInfoDir = ensureDirExists(buildPath(trashBasePath, "info"));
292             string trashFilePathsDir = ensureDirExists(buildPath(trashBasePath, "files"));
293 
294             string trashInfoPath = buildPath(trashInfoDir, path.baseName ~ ".trashinfo");
295             string trashFilePath = buildPath(trashFilePathsDir, path.baseName);
296 
297             import std.datetime;
298             import std.conv : octal;
299             import core.stdc.errno;
300 
301             auto currentTime = Clock.currTime;
302             currentTime.fracSecs = Duration.zero;
303             string timeString = currentTime.toISOExtString();
304             string contents = format("[Trash Info]\nPath=%s\nDeletionDate=%s\n", (usingTopdir ? path.relativePath(fileTopDir) : path).encode(), timeString);
305 
306             const mode = O_CREAT | O_WRONLY | O_EXCL;
307             int fd;
308             uint number = 1;
309             while(trashFilePath.exists || ((fd = .open(toStringz(trashInfoPath), mode, octal!666)) == -1 && errno == EEXIST)) {
310                 string baseName = numberedBaseName(path, number);
311                 trashFilePath = buildPath(trashFilePathsDir, baseName);
312                 trashInfoPath = buildPath(trashInfoDir, baseName ~ ".trashinfo");
313                 number++;
314             }
315             errnoEnforce(fd != -1);
316             scope(exit) .close(fd);
317             errnoEnforce(write(fd, contents.ptr, contents.length) == contents.length);
318 
319             path.rename(trashFilePath);
320         } else {
321             static assert("Unsupported platform");
322         }
323     }
324 }

424     static @trusted string StrRetToString(ref STRRET strRet)
425     {
426         switch (strRet.uType)
427         {
428         case STRRET_CSTR:
429             return fromStringz(strRet.cStr.ptr).idup;
430         case STRRET_OFFSET:
431             return string.init;
432         case STRRET_WSTR:
433             char[MAX_PATH] szTemp;
434             auto len = WideCharToMultiByte (CP_ACP, 0, strRet.pOleStr, -1, szTemp.ptr, szTemp.sizeof, null, null);
435             scope(exit) CoTaskMemFree(strRet.pOleStr);
436             if (len)
437                 return szTemp[0..len-1].idup;
438             else
439                 return string.init;
440         default:
441             return string.init;
442         }
443     }

536     private static struct ByItem
537     {
538         this(IShellFolder folder) {
539             this.folder = folder;
540             folder.AddRef();
541             with(SHCONTF) henforce(folder.EnumObjects(null, SHCONTF_FOLDERS | SHCONTF_NONFOLDERS | SHCONTF_INCLUDEHIDDEN, &enumFiles), "Failed to enumerate objects in recycle bin");
542             popFront();
543         }
544         this(this) {
545             if (enumFiles)
546                 enumFiles.AddRef();
547             if (folder)
548                 folder.AddRef();
549         }
550         ~this() {
551             if (enumFiles)
552                 enumFiles.Release();
553             if (folder)

560             import std.algorithm.mutation : move;

564             LPITEMIDLIST pidl;
565             if (enumFiles.Next(1, &pidl, null) == S_FALSE) {

572             }
573         }

575             return atTheEnd;
576         }

602     final class Trashcan : ITrashcan
603     {
604         private @safe static bool isDirNothrow(string path) nothrow {
605             bool isDirectory;
606             if (collectException(path.isDir, isDirectory) is null)
607                 return isDirectory;
608             return false;
609         }
610 
611         @safe this() {
612         }
613 
614         import std.typecons : Tuple;
615 
616         @trusted InputRange!TrashcanItem byItem() {
617             import std.algorithm.iteration : cache, map, joiner, filter;
618             alias Tuple!(string, "base", string, "info", string, "files", string, "root") TrashLocation;
619             return inputRangeObject(standardTrashBasePaths().map!(trashDir => TrashLocation(trashDir.base, buildPath(trashDir.base, "info"), buildPath(trashDir.base, "files"), trashDir.root)).filter!(t => isDirNothrow(t.info) && isDirNothrow(t.files)).map!(delegate(TrashLocation trash) {
620                 InputRange!TrashcanItem toReturn;
621                 try {
622                     toReturn = inputRangeObject(dirEntries(trash.info, SpanMode.shallow, false).filter!(entry => entry.extension == ".trashinfo").map!(delegate(DirEntry entry) {
623                         string trashedFile = buildPath(trash.files, entry.baseName.stripExtension);
624                         try {
625                             if (exists(trashedFile)) {
626                                 import inilike.read;
627 
628                                 string path;
629 
630                                 auto onLeadingComment = delegate void(string line) {};
631                                 auto onGroup = delegate ActionOnGroup(string groupName) {
632                                     if (groupName == "Trash Info")
633                                         return ActionOnGroup.stopAfter;
634                                     return ActionOnGroup.skip;
635                                 };
636                                 auto onKeyValue = delegate void(string key, string value, string groupName) {
637                                     if (groupName == "Trash Info" && key == "Path")
638                                         path = value;
639                                 };
640                                 auto onCommentInGroup = delegate void(string line, string groupName) {};
641                                 readIniLike(iniLikeFileReader(entry.name), onLeadingComment, onGroup, onKeyValue, onCommentInGroup);
642 
643                                 if (path.length) {
644                                     path = path.decode();
645                                     string restorePath;
646                                     if (path.isAbsolute)
647                                         restorePath = path;
648                                     else
649                                         restorePath = buildPath(trash.root, path);
650                                     return TrashcanItem(restorePath, trashedFile.isDir, entry.name, trashedFile);
651                                 }
652                             }
653                         } catch(Exception e) {}
654                         return TrashcanItem.init;
655                     }).cache.filter!(item => item.restorePath.length));
656                 } catch(Exception e) {
657                     toReturn = inputRangeObject(TrashcanItem[].init);
658                 }
659                 return toReturn;
660             }).cache.joiner);
661         }
662 
663         @safe void restore(ref scope TrashcanItem item) {
664             rename(item.trashedPath, item.restorePath);
